SELECT        
a.ID
, a.LVS_ID
, a.Brutto
, d.KD_ID
, a.Werk
, a.Halle
, a.Reihe
, a.Ebene
, a.Platz
, CASE
	WHEN (a.Dicke>0) AND (a.Breite>0) AND (a.Laenge>0) AND (a.Hoehe>0) 
		THEN CAST(a.Dicke as varchar)+' x '+ CAST(a.Breite as varchar)+' x '+CAST(a.Laenge as varchar)+' x '+CAST(a.Hoehe as varchar)
	WHEN (a.Dicke>0) AND (a.Breite>0) AND (a.Laenge>0) AND (a.Hoehe=0) 
		THEN CAST(a.Dicke as nvarchar) +' x '+ CAST(a.Breite as nvarchar)+' x '+CAST(a.Laenge as nvarchar)
	WHEN (a.Dicke>0) AND  (a.Breite>0) AND (a.Laenge=0) AND (a.Hoehe=0) 
		THEN CAST(a.Dicke as varchar)+' x '+ CAST(a.Breite as varchar)
	WHEN (a.Dicke>0) AND (a.Breite=0) AND (a.Laenge=0) AND (a.Hoehe=0) 
		THEN CAST(a.Dicke as varchar)
	ELSE ''
	END as Abmessung

FROM Artikel a
INNER JOIN LEingang c ON c.ID = a.LEingangTableID
INNER JOIN ADR d ON d.ID=c.Auftraggeber
WHERE a.LEingangTableID=@LEingangTableID
